type User @model {
  id: ID!
  username: String!
  email: String!
  emailVerified: Boolean
  notes: [Note] @connection(name: "UserNotes")
}

type Note @model {
  id: ID!
  createdAt: String
  longitude: String
  latitude: String
  altitude: String
  content: String
  user: User @connection(name: "UserNotes")
}
